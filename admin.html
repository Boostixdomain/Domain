async function addProduct() { const name = document.getElementById("name").value; const price = document.getElementById("price").value; const file = document.getElementById("imageFile").files[0]; if (!file) { alert("Select image"); return; } const storageRef = storage.ref("products/" + file.name); await storageRef.put(file); const imageURL = await storageRef.getDownloadURL(); await db.collection("products").add({ name: name, price: price, image: imageURL }); alert("Product uploaded 🚀"); }
